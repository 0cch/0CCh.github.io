.if (${/d:$VectorType}) {
	r? $t0 = @@C++(*((${$VectorType} *)@$t0))
}

.if (${/d:$arg2}) { 
    .if ($sicmp("${$arg2}", "-c") == 0) { 
        r $t2 = 0 
        aS ${/v:command} "${$arg3}" 
    } 
}
.else { 
    r $t2 = 1 
    aS ${/v:command} " " 
}


r? $t1 = @@C++(@$t0._Mylast)

.while (@$t0 != @$t1) {
	.if ($t2 == 1) {
		?? @@c++(@$t0->_Bx)

}

        .push /r /q 
        r? $t0 = @$t0->_Left 
        $$>a< ${$arg0} -n 
        .pop /r /q 
    } 
	
	.if (@@C++(@$t0->_Isnil) == 0) { 
        .if (@$t2 == 1) { 
            .printf /D "%p\n", @$t0, @$t0 
            .printf "key = " 
            ?? @$t0->_Myval.first 
            .printf "value = " 
            ?? @$t0->_Myval.second 
        }
		.else { 
            r? $t9 = @$t0->_Myval 
            command 
        } 
    }
	
    .if (@@C++(@$t0->_Right) != @@C++(@$t1)) { 
        .push /r /q 
        r? $t0 = @$t0->_Right 
        $$>a< ${$arg0} -n 
        .pop /r /q 
    }
	
}
